#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    
    int min_x=dots[0][0],max_x=dots[0][0];
    int min_y=dots[0][1],max_y=dots[0][1];
    
    for(int i=1;i<dots.size();i++)
    {
        if(dots[i][0]<min_x)
            min_x=dots[i][0];
        if(dots[i][0]>max_x)
            max_x=dots[i][0];
        if(dots[i][1]<min_y)
            min_y=dots[i][1];
        if(dots[i][1]>max_y)
            max_y=dots[i][1];
    }
    
    answer=(max_x-min_x)*(max_y-min_y);
    
    return answer;
}
