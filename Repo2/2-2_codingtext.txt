#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    int m=board.size();
    int n=board[0].size();
    
    for(int i=0;i<m;i++)
    {
        for (int j=0;j<n;j++)
        {
            if(i+j<=k){
                answer=answer+board[i][j];
            }
        }
    }
    
    return answer;
}
